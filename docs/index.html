Швидкий старт

# 1. Клонувати та встановити
git clone https://github.com/AndriiHubanov/legal-agents.git
cd legal-agents
python -m venv venv && venv\Scripts\activate
pip install -r requirements.txt
playwright install chromium
# 2. Налаштувати .env
copy .env.example .env
# Додати ANTHROPIC_API_KEY у .env
# 3. Зібрати рішення
python main.py collect --category civil --date-from 2023-01-01 --keywords "оренда,земля"

# 4. Проаналізувати справу
python main.py analyze --case-file case_example.json

# 5. Згенерувати документ
python main.py generate --analysis-file report.json --doc-type appeal --case-number "757/123/24-ц"
